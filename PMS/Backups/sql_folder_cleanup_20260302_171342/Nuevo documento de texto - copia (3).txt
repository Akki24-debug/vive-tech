DROP PROCEDURE IF EXISTS sp_list_reservations_by_property $$
CREATE PROCEDURE sp_list_reservations_by_property
(
  IN p_property_id   INT,
  IN p_property_code VARCHAR(32)
)
proc:BEGIN
  DECLARE v_id_property INT;

  IF p_property_id IS NOT NULL AND p_property_id <> 0 THEN
    SET v_id_property = p_property_id;
  ELSE
    SELECT id_property INTO v_id_property
    FROM property
    WHERE code = p_property_code
    LIMIT 1;
  END IF;

  IF v_id_property IS NULL THEN
    SELECT 'ERROR' AS status, 'Unknown property (id/code)' AS message; LEAVE proc;
  END IF;

  SELECT
    g.names          AS guest_names,
    g.last_name      AS guest_last_name,
    p.name           AS hospedaje,
    r.check_in_date  AS check_in,
    r.check_out_date AS check_out,
    rm.code          AS room_code
  FROM reservation r
  JOIN guest g   ON g.id_guest = r.id_guest
  JOIN property p ON p.id_property = r.id_property
  JOIN room rm   ON rm.id_room = r.id_room
  WHERE r.id_property = v_id_property
  ORDER BY r.check_in_date, r.id_reservation;
END $$

DELIMITER ;
